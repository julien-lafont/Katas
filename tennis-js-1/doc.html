A game is played with two players

		var game = tennis.game("Julien", "Marc");
		(game.players[0].name).should.equal("Julien");
		(game.players[1].name).should.equal("Marc");
A the start of the game every player has 0 point

		newTestGame().forEachPlayer(function(player) {
			(player.score).should.equal(0);
		})
When a player won a point, his score is 15

		var game = newTestGame();
		game.playerWin(0);
		(game.players[0].score).should.equal(15);
When a player won 4 times, he win the match

		var game = newTestGame();
		for (var i in fourTimes) game.playerWin(0);
		(game.winner).should.equal(0);
When a player won 3 times and the opposent 4 times, there is no Winner and score are 40/40

		var game = initDeuceGame();
		(game.winner).should.not.be.ok;
		game.forEachPlayer(function(player) { (player.score).should.equal(40); });
.. And the second player has the Advantage

		var game = initDeuceGame();
		(game.players[1].advantage).should.be.true;
		(game.players[0].advantage).should.be.false;
If the player who has the advantage won a new point, he is the Winner

		var game = initDeuceGame();
		game.playerWin(1);
		(game.winner).should.equal(1);
If a player who have the advatange lost the point, he lost his advatange

		var game = initDeuceGame();
		game.playerWin(0);
		(game.players[1].advantage).should.be.false;
If a player who have the advantage lost the point, his opponent doesn't have the advantage

		var game = initDeuceGame();
		game.playerWin(0);
		(game.players[0].advantage).should.be.false;
		(game.winner).should.not.be.ok;
There is a winner after many deuce situations

		var game = initDeuceGame();
		game.playerWin(0);
		game.playerWin(1);
		game.playerWin(1);
		(game.winner).should.equal(1);
It's not possible to win a point whe the match is ended
